#function declaration

def S(n):
    if n==0:
        return 1
    else:
        return 2*S(n-1)
number =int(input("Enter number: "))

#Iterate until user input -1 as input

while number!= -1:
    getValue= S(number)
    answer= getValue-1
    print("Output: ",answer)
    number = int(input("Enter number: "))
else:
    print("Output: Finished")
